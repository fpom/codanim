\def\hl #1{{\setbox0=\hbox{#1}%
    \adjustbox{set vsize={\ht0}{\dp0}}{\hbox to \wd0{\hss\colorbox{yellow}{#1}\hss}}}}

\def\hl #1{%
  \tikz[overlay,baseline,inner sep=1pt]
    \node[fill=yellow,anchor=text] {\color{yellow}#1};%
  #1%
}

\def\onlyhl #1{\only<#1>\hl}
